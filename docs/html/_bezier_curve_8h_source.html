68  std::array<double, 2> evaluate(double t) const
69  {
70  const std::vector<std::array<double, 2>> &P = m_points;
71  const int N = degree();
72 
73  // Check for at least two anchor points
74  if (N < 1)
75  {
76  return {0,0};
77  }
78 
79  double x = 0, y = 0;
80  for(int i = 0; i <= N; i++)
81  {
82  x += binomialCoefficient(N, i) * pow(1 - t, N - i) * pow(t, i) * P[i][0];
83  y += binomialCoefficient(N, i) * pow(1 - t, N - i) * pow(t, i) * P[i][1];
84  }
85  return {x, y};
86  }

104  {

107 
108  double x = (P0[0] + P1[0]) / 2; // P0[0] + (P1[0] - P0[0])*0.5;
109  double y = (P0[1] + P1[1]) / 2; // P0[1] + (P1[1] - P0[1])*0.5;
