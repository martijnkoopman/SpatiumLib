68  void lookAt(const geom3d::Point3 &target, const geom3d::Vector3 &up)
69  {
70  // Compute vector from target to eye
71  geom3d::Vector3 back = geom3d::Vector3(m_transform.position() - target).normalized();
72 
73  // Normalize and orthogonalize view up vector
74  geom3d::Vector3 right = up.normalized().cross(back);
75  geom3d::Vector3 upOrtho = back.cross(right);
76 
77  // Set rotation of transformation matrix based on view direction and view
78  // up vector.
79  geom3d::Matrix4x4 M = m_transform.matrix();
80 
81  // First column is right vector
82  M(0,0) = right(0);
83  M(1,0) = right(1);
84  M(2,0) = right(2);
85 
86  // Second column is up vector
87  M(0,1) = upOrtho(0);
88  M(1,1) = upOrtho(1);
89  M(2,1) = upOrtho(2);
90 
91  // Third column is back vector
92  M(0,2) = back(0);
93  M(1,2) = back(1);
94  M(2,2) = back(2);
95 
96  m_transform.setMatrix(M);
97  }
