62 inline int solveQuadratic(double a, double b, double c, double &x1, double &x2)
63 {
64  // Compute discriminant
65  double D = (b * b) - (4 * a * c);
66  if (D < 0)
67  {
68  return 0;
69  }
70

73  x2 = (-b + sqrt(D)) / (2 * a);

102 inline int solveCubic(double a, double b, double c, double d, double &x1, double &x2, double &x3)
103 {
104  // http://www2.trinity.unimelb.edu.au/~rbroekst/MathX/Cubic%20Formula.pdf
105  // https://en.wikibooks.org/wiki/Trigonometry/The_solution_of_cubic_equations
106 
107  // Make a = 1. Then omit d
108  if (a != 1)
109  {
110  b /= a;
111  c /= a;
112  d /= a;
113  }
114  a = b; b = c; c = d; d = 0;
115 
116  double p = b - ((a*a) / 3);
117  double q = ((2*(a*a*a)) / 27) - ((a*b) / 3) + c;
118  //double D = ((q*q) / 4) + ((p*p*p) / 27);
119 
120  double Q = (3*b - a*a) / 9;
121  double R = (9*a*b - 27*c - 2*a*a*a) / 54;
122  double D = Q*Q*Q + R*R;
123 
124  if (D < std::numeric_limits<double>::epsilon()
125  && D > -std::numeric_limits<double>::epsilon()) // D == 0
126  {
127  // There are TWO real solutions
128  x1 = -2*cbrt(q/2) - (a/3);
129  x2 = cbrt(q/2) - (a/3);
130 
131  // Reorder results
132  if (x1 > x2)
133  {
134  std::swap(x1, x2);
135  }
136  x3 = x2;
137 
138  return 2;
139  }
140  else if (D > 0)
141  {
142  // There is ONE real solution
143  x1 = x2 = x3 = cbrt(-1*(q/2) + sqrt(D)) + cbrt(-1*(q/2) - sqrt(D)) - (a/3);
144 
145  return 1;
146  }
147  else if(D < 0)
148  {
149  // There are THREE real solution
150 
151  // Solve with trigonometry
152  double theta = acos(R/sqrt(-(Q*Q*Q)));
153  x1 = 2 * sqrt(-Q) * cos(theta / 3) - (a/3);
154  x2 = 2 * sqrt(-Q) * cos(theta / 3 + (2*PI/3)) - (a/3);
155  x3 = 2 * sqrt(-Q) * cos(theta / 3 + (4*PI/3)) - (a/3);
156 
157  // Reorder results
158  if (x1 > x2)
159  {
160  std::swap(x1, x2);
161  }
162  if (x1 > x3)
163  {
164  std::swap(x1, x3);
165  }
166  if (x2 > x3)
167  {
168  std::swap(x2, x3);
169  }
170 
171  return 3;
172  }
173 }
