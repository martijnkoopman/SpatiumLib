83  static PointQuadtree buildFromPoints(const std::vector<std::array<double, 2>> &points, size_t maxPointCountLeaf = 100)
84  {
85  if (points.size() == 0)
86  {
87  BoundingBox<double,2> b;
88  PointQuadtree p(b);
89  return p;
90  }
91 
92  // Compute minimum bounding rectangle
93  BoundingBox<double,2> bounds = BoundingBox<double,2>::fromPoints(points);
94 
95  // Derive minimum bounding square
96  double width = bounds.diameter(0);
97  double height = bounds.diameter(1);
98  std::array<double, 2> min = bounds.min();
99  std::array<double, 2> max = bounds.max();
100  if (width > height) {
101  max = { min[0] + width, min[1] + width };
102  } else {
103  max = { min[0] + height, min[1] + height };
104  }
105  bounds = BoundingBox<double,2>::fromMinMax(min, max);
106 
107  // Create inital tree
108  PointQuadtree tree(bounds, maxPointCountLeaf);
109 
110  // Add points
111  for(auto point : points)
112  {
113  tree.addPoint(point);
114  }
115 
116  return tree;
117  }

128  bool addPoint(const std::array<double, 2> &point)
129  {
130  if (!m_boundingBox.isInside(point))
131  {
132  return false;
133  }
134 
135  std::shared_ptr<PointQuadtreeNode> node = std::static_pointer_cast<PointQuadtreeNode>(this->m_root);
136  BoundingBox<double,2> bounds = m_boundingBox;
137  while(node != nullptr)
138  {
139  // Check: Is leaf node?
140  if (node->hasChildren())
141  {
142  // No leaf node. Pass to one of children.
143  // Determine child and traverse into
144  size_t childIndex = PointQuadtree::determineChild(bounds, point, bounds);
145  node = std::static_pointer_cast<PointQuadtreeNode>(node->child(childIndex));
146  }
147  else
148  {
149  // Leaf node. Add point. May need splitting child.
150  if (node->object().size() + 1 > m_maxPointCountLeaf)
151  {
152  // Split node. Create 4 children.
153  node->createChildren(node);
154 
155  // Distribute points of current node over 4 new children
156  for (size_t i = 0; i < node->object().size(); i++)
157  {
158  std::array<double, 2> childPoint = node->object()[i];
159 
160  BoundingBox<double,2> curBounds;
161  size_t curChildIndex = PointQuadtree::determineChild(bounds, childPoint, curBounds);
162  std::shared_ptr<PointQuadtreeNode> curChild = std::static_pointer_cast<PointQuadtreeNode>(node->child(curChildIndex));
163  curChild->object().push_back(childPoint);
164  }
165  node->object().clear();
166 
167  // Determine next child and traverse into
168  size_t childIndex = PointQuadtree::determineChild(bounds, point, bounds);
169  node = std::static_pointer_cast<PointQuadtreeNode>(node->child(childIndex));
170  }
171  else
172  {
173  // Simply add point
174  node->object().push_back(point);
175 
176  // Stop tree traversal
177  node = nullptr;
178  }
179  }
180  }
181 
182  return true;
183  }

192  static size_t determineChild(const BoundingBox<double,2> &boundsParent, const std::array<double, 2> &point, BoundingBox<double,2> &boundsChild)
193  {
194  std::array<double, 2> min = boundsParent.min();
195  std::array<double, 2> max = boundsParent.max();
196  std::array<double, 2> center = boundsParent.center();
197  if (point[0] < center[0])
198  {
199  // Left half
200  if (point[1] < center[1])
201  {
202  // Bottom
203  boundsChild = BoundingBox<double,2>::fromMinMax(min, center);
204  return 0;
205  }
206  else
207  {
208  // Top
209  boundsChild = BoundingBox<double,2>::fromMinMax({ min[0], center[1] }, { center[0], max[1] });
210  return 2;
211  }
212  }
213  else
214  {
215  // Right half
216  if (point[1] < center[1])
217  {
218  // Top
219  boundsChild = BoundingBox<double,2>::fromMinMax({ center[0], min[1] }, { max[0], center[1] });
220  return 1;
221  }
222  else
223  {
224  // Bottom
225  boundsChild = BoundingBox<double,2>::fromMinMax(center, max);
226  return 3;
227  }
228  }
229  }
