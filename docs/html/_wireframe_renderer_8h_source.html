50  bool render(const Scene &scene, Image<unsigned char, 3> &image) override
51  {
52  // Check for camera in scene
53  auto camera = scene.camera();
54  if (camera == nullptr)
55  {
56  return false;
57  }
58 
59  // For now only support orthographic projection
60  std::shared_ptr<OrthographicCamera> cameraOrtho = std::dynamic_pointer_cast<OrthographicCamera>(camera);
61  if (cameraOrtho == nullptr)
62  {
63  return false;
64  }
65 
66  // 1. Construct view port matrix based on image resolution.
67  double w = static_cast<double>(image.width());
68  double h = static_cast<double>(image.height());
69 
70  geom3d::Matrix4x4 matrixViewport({ { w/2, 0 , 0, (w-1)/2 },
71  { 0 , h/2, 0, (h-1)/2 },
72  { 0 , 0 , 1, 0 },
73  { 0 , 0 , 0, 1 } });
74 
75 
76  // 2. Construct orthographic projection matrix based on canonical view volume.
77  const double n = -cameraOrtho->near();
78  const double f = -cameraOrtho->far();
79  const double r = cameraOrtho->size() * 0.5 * w/h; // w/h = aspect ratio
80  const double l = -r;
81  const double t = cameraOrtho->size() * 0.5;
82  const double b = -t;
83 
84  geom3d::Matrix4x4 matrixOrtho({ { 2/(r-l), 0 , 0 , -(r+l)/(r-l) },
85  { 0 , 2/(t-b), 0 , -(t+b)/(t-b) },
86  { 0 , 0 , 2/(n-f), -(n+f)/(n-f) },
87  { 0 , 0 , 0 , 1 } });
88 
89  // Iterate objects in scene
90  for (auto renderObject : scene.renderObjects())
91  {
92  // For now only support meshes
93  std::shared_ptr<Mesh> mesh = std::dynamic_pointer_cast<Mesh>(renderObject);
94  if (mesh)
95  {
96  // 3. Construct camera transformation matrix (World space -> Camera space)
97  geom3d::Matrix4x4 matrixCamera = cameraOrtho->transform().matrix().inverse();
98 
99  // 4. Construct world transformation matrix (object space -> world space)
100  geom3d::Matrix4x4 matrixObject = mesh->transform().matrix();
101 
102  // Combine matrices to form a matrix that projects a point from local
103  // object space to screen space.
104  geom3d::Matrix4x4 matrixPipeline = matrixViewport * matrixOrtho *
105  matrixCamera * matrixObject;
106 
107  // Iterate edges in object
108  for (auto edge : mesh->edges())
109  {
110  // Get vertices of edge
111  geom3d::Point3 v1 = mesh->vertex(edge[0]);
112  geom3d::Point3 v2 = mesh->vertex(edge[1]);
113 
114  // Project vertices to screen space
115  geom3d::Point3 v1Proj = matrixPipeline * v1;
116  geom3d::Point3 v2Proj = matrixPipeline * v2;
117 
118  std::cout << v1 << " to " << v1Proj << std::endl;
119  std::cout << v2 << " to " << v2Proj << std::endl;
120 
121  int x1 = static_cast<int>(v1Proj.x());
122  int y1 = static_cast<int>(v1Proj.y());
123  int x2 = static_cast<int>(v2Proj.x());
124  int y2 = static_cast<int>(v2Proj.y());
125

127

129  }
